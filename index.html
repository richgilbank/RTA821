    The problem

    
var years = [];

for (var i = 0; i <= 9; i++) {
    years[i] = function () {
        console.log("The year is 201" + i);
    };
}

years[4]();
        
    
// The year is 2019
// ಠ_ಠ

    The Solution

    
var years = [];

for (let i = 0; i <= 9; i++) {
    years[i] = function () {
        console.log("The year is 201" + i);
    };
}

years[4]();
        
    
// The year is 2014
